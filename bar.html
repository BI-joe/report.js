ChartJS Bar example

var renderer = new reportjs.Renderer(),
    options = {
        data: {
            dimensions: [{ id: 'Measures'}, { id: 'Year' }, { id: 'Month' }],
            dimensionValues: [
                [{ id: 'count', caption: 'My Count' }],
                [{ id: '2013' }, { id: '2014' }],
                [{ id: 'January' }, { id: 'February' }, { id: 'March' }]
            ],
            cells: [
                {value: 10, dimensionValues: [0, 0, 0]},
                {value: 15, dimensionValues: [0, 0, 1]},
                {value: 9, dimensionValues: [0, 0, 2]},
                {value: 10, dimensionValues: [0, 1, 0]},
                {value: 15, dimensionValues: [0, 1, 1]},
                {value: 12, dimensionValues: [0, 1, 2]}
            ]
        },
        layout: {
            type: 'graph',
            labels: ['Year', 'Month'],
            datasets: ['Measures'],
            graphType: 'bar'
        }
    };

renderer.renderTo($('#mydiv'), options);
